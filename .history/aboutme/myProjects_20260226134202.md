# Project #1
This was the final project for programming, we were tasked to make a game in C# that was unique and fun to play, for this project/game I chose to make a game where you cut grass and you could unlock some powerups inside of the game when you would walk over the grass which would help you cut grass easier. 
```
using System;

namespace Final_game
{
    class Program
    {
        static void Main()
        {
            Console.SetWindowSize(50, 20);
            Console.CursorVisible = false;

            // Set up the game field dimensions
            int width = 45;
            int height = 15;
            int playerX = width / 2;
            int playerY = height / 2;

            // Track which grass has been cut and where items are
            bool[,] grassCut = new bool[width, height];
            string[,] items = new string[width, height];

            // Place random items on the field
            Random random = new Random();

            // Place 8 powerup items
            for (int i = 0; i < 8; i++)
                items[random.Next(width), random.Next(height)] = "!";

            // Place 5 treasure items
            for (int i = 0; i < 5; i++)
                items[random.Next(width), random.Next(height)] = "$";

            // Game state variables
            int powerups = 0;
            int treasures= 0;
            bool hasPowerup = false;
            int powerupTimer = 100;
            bool win = false;

            // Display game instructions
            Console.WriteLine("Grass Cutter Game!");
            Console.WriteLine("Use W/A/S/D to move and cut grass");
            Console.WriteLine("Find hidden items: ! powerups, $ treasures");
            Console.WriteLine("Press any key to start...");
            Console.ReadKey();

            // Main game loop
            while (!win)
            {
                Console.SetCursorPosition(0, 0);

                // Check if player found an item on uncut grass 
                if (!grassCut[playerX, playerY] && items[playerX, playerY] != null)
                {
                    if (items[playerX, playerY] == "!")
                    {
                        powerups++;
                        hasPowerup = true;
                        powerupTimer = 100;
                    }
                    else if (items[playerX, playerY] == "$") treasures++;
                }

                // Mark current position as cut
                grassCut[playerX, playerY] = true;

                // If powerup is active then the grass be getting cut in larger area
                if (hasPowerup && powerupTimer > 0)
                {
                    for (int offsetX = -1; offsetX <= 2; offsetX++)
                    {
                        for (int offsetY = -2; offsetY <= 2; offsetY++)
                        {
                            int targetX = playerX + offsetX;
                            int targetY = playerY + offsetY;

                            // Make sure we stay tha boundaries
                            if (targetX >= 0 && targetX < width &&
                                targetY >= 0 && targetY < height)
                            {
                                grassCut[targetX, targetY] = true;
                            }
                        }
                    }
                    powerupTimer--;
                    if (powerupTimer == 0) hasPowerup = false;
                }

                // Draw the game field
                for (int y = 0; y < height; y++)
                {
                    for (int x = 0; x < width; x++)
                    {
                        if (x == playerX && y == playerY) Console.Write("*");
                        else if (grassCut[x, y]) Console.Write(items[x, y] ?? " ");
                        else Console.Write("░");
                    }
                    Console.WriteLine();
                }

                // Count remaining grass
                int grassLeft = 0;
                for (int y = 0; y < height; y++)
                    for (int x = 0; x < width; x++)
                        if (!grassCut[x, y]) grassLeft++;

                // Display game statistics
                Console.WriteLine();
                Console.WriteLine($"Grass left: {grassLeft} | Powerups: {powerups} | Treasure Coins: {treasures}");
                Console.WriteLine(hasPowerup ? $"POWERUP ACTIVE! Time left: {powerupTimer}" : "Use W/A/S/D to move");

                // Check win condition
                if (grassLeft == 0)
                {
                    win = true;
                    Console.WriteLine("You win! All grass is cut!");
                    Console.WriteLine($"Final score - Powerups: {powerups} Treasures: {treasures}");
                    Console.ReadLine();
                }

                // Handle player input
                if (Console.KeyAvailable)
                {
                    char input = Console.ReadKey(true).KeyChar;
                    if ((input == 'w' || input == 'W') && playerY > 0) playerY--;
                    else if ((input == 's' || input == 'S') && playerY < height - 1) playerY++;
                    else if ((input == 'a' || input == 'A') && playerX > 0) playerX--;
                    else if ((input == 'd' || input == 'D') && playerX < width - 1) playerX++;
                }
            }
        }
    }
}
```
